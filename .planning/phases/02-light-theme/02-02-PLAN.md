    ```typescript
    // Add after existing refs:
    const theme = ref<Theme>('dark');
    ```

    2. Add initTheme function (runs on store initialization):
    ```typescript
    function initTheme() {
      const saved = localStorage.getItem('projectflow-theme');
      if (saved === 'light' || saved === 'dark') {
        theme.value = saved;
      } else if (window.matchMedia('(prefers-color-scheme: light)').matches) {
        theme.value = 'light';
      }
      document.documentElement.setAttribute('data-theme', theme.value);
    }
    ```

    3. Add toggleTheme function:
    ```typescript
    function toggleTheme() {
      theme.value = theme.value === 'dark' ? 'light' : 'dark';
      document.documentElement.setAttribute('data-theme', theme.value);
      localStorage.setItem('projectflow-theme', theme.value);
    }
    ```

    4. Export the new state and functions:
    ```typescript
    return {
      // ... existing exports
      theme,
      initTheme,
      toggleTheme,
    };
    ```

    5. Call initTheme() when the store is first used - add this at the end of defineStore:
    ```typescript
    // Initialize theme on store creation
    initTheme();
    ```
