3. Add Anthropic client to LitellmClient struct:

4. Initialize Anthropic client in new() for Anthropic models:


5. Implement stream_chat for Anthropic using SDK:
```rust
if let Some(ref client) = self.anthropic_client {
    let request = CreateMessageRequest {
        model: self.model.clone(),
        messages: messages.into_iter().map(|m| {
            let role = match m.role.as_str() {
                "system" => Role::System,
                "assistant" => Role::Assistant,
                _ => Role::User,
            };
            anthropic::types::Message::new(role, m.content)
        }).collect(),
        max_tokens: 4096,
        stream: true,
        ..Default::default()
    };

    let stream = client.messages().stream(request).await.map_err(|e| e.to_string())?;
    // Convert MessageStreamEvent to String stream
    let stream = stream.map(|event| {
        match event {
            Ok(MessageStreamEvent::ContentBlockDeltaEvent { delta, .. }) => {
                Ok(delta.text.clone())
            }
            Ok(MessageStreamEvent::MessageStopEvent { .. }) => {
                Ok(String::new()) // Signal end
            }
            Err(e) => Err(e.to_string()),
            _ => Ok(String::new()),
        }
    }).boxed();
    return Ok(stream);
}
```
